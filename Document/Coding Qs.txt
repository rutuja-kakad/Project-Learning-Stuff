   printf("Hello World!!!!");
   char src[10]="Apple";
   char dest[10];
   int i,j;
   i=0,j=0;
   while(src[j]!='\0')
   {
       j++;
   }
    
    printf("\nString length:%d",j);
    j--;
    while(src[i]!='\0')
    {
        dest[i]=src[j];
        j--;
        i++;
    }
    
    printf("\nString Before Reverse : %s",src);
    printf("\nReversed After String : %s", dest);
    
    return 0;
}

#include <stdio.h>
int main()
{
    int i,j,n;
    printf("Enter the no of rows :");
    scanf("%d",&n);
    for(i=0;i<=n;i++)
    {
        for(j=0;j<i;j++)
        {
            printf("* ");
        }
        printf("\n");
